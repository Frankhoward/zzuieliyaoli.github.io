##一、
    var name = "The Window";
	var object = {
		name: "The Object",
		getName: function(){
			return function(){        
				return this.name	  
			};
		}
	};
	console.log(object.getName()()); // "The Window"


	var name = "The Window";
	var object = {
		name: "The Object",
		getName: function(){
			return function(){        
				alert(this.name);     
			}();
		}
	};
	object.getName(); // "The Window"
